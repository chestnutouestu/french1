function getPP(inf){if(PP_IRREG[inf])return PP_IRREG[inf];if(inf.endsWith('er'))return inf.slice(0,-2)+'é';if(inf.endsWith('ir'))return inf.slice(0,-2)+'i';if(inf.endsWith('re'))return inf.slice(0,-2)+'u';return inf.slice(0,-2)+'i';}
function getAux(inf){if(inf.startsWith('se ')||inf.startsWith("s'"))return 'être';return ETRE_VERBS.has(inf)?'être':'avoir';}
function auxForm(aux,t,i){return IRREG[aux]?.[t]?.[i]||aux;}
function etreAgree(pp,pron){if(!pp||pp==='—')return pp;const fem=/elle/.test(pron)&&!/elles/.test(pron),plFem=/elles/.test(pron),pl=/nous|vous|ils/.test(pron)||plFem;if(plFem)return pp+'es';if(fem)return pp+'e';if(pl)return pp+'s';return pp;}
function subjImpf(ps){const b=ps[0];if(!b||b==='—')return Array(6).fill('—');const st=b.endsWith('s')?b.slice(0,-1):b;const last=st.slice(-1),rest=st.slice(0,-1),acc={a:'â',e:'ê',i:'î',o:'ô',u:'û'};return[st+'sse',st+'sses',rest+(acc[last]||last)+'t',st+'ssions',st+'ssiez',st+'ssent'];}

function fillCompound(c,inf){
  const pp=getPP(inf),aux=getAux(inf);
  const pc6=['je','tu','il / elle','nous','vous','ils / elles'];
  const pq6=["j'",'tu','il / elle','nous','vous','ils / elles'];
  if(!c.passé_composé)c.passé_composé=pc6.map((p,i)=>auxForm(aux,'présent',i)+' '+(aux==='être'?etreAgree(pp,p):pp));
  if(!c.plus_que_parfait)c.plus_que_parfait=pq6.map((p,i)=>auxForm(aux,'imparfait',i)+' '+(aux==='être'?etreAgree(pp,p):pp));
  if(!c.futur_antérieur)c.futur_antérieur=pq6.map((p,i)=>auxForm(aux,'futur_simple',i)+' '+(aux==='être'?etreAgree(pp,p):pp));
  if(!c.conditionnel_passé)c.conditionnel_passé=pq6.map((p,i)=>auxForm(aux,'conditionnel_présent',i)+' '+(aux==='être'?etreAgree(pp,p):pp));
  if(!c.subjonctif_imparfait&&c.passé_simple)c.subjonctif_imparfait=subjImpf(c.passé_simple);
  if(!c.impératif_présent&&c.présent)c.impératif_présent=[c.présent[1],c.présent[3],c.présent[4]];
  return c;
}

function buildCond(fut){return fut.map((f,i)=>{const sfx=['ai','as','a','ons','ez','ont'][i];return f.slice(0,-sfx.length)+['ais','ais','ait','ions','iez','aient'][i];});}
function erPres(inf){const s=inf.slice(0,-2);if(inf.endsWith('ger'))return[s+'e',s+'es',s+'e',s+'eons',s+'ez',s+'ent'];if(inf.endsWith('cer')){const sc=s.slice(0,-1)+'ç';return[s+'e',s+'es',s+'e',sc+'ons',s+'ez',s+'ent'];}if(inf.endsWith('eler')||inf.endsWith('eter')){const s2=s.slice(0,-1),ll=s.slice(-1);return[s2+ll+ll+'e',s2+ll+ll+'es',s2+ll+ll+'e',s+'ons',s+'ez',s2+ll+ll+'ent'];}if(/é[^aeiou]+er$/.test(inf)){const ss=s.replace(/é([^aeiou]+)$/,'è$1');return[ss+'e',ss+'es',ss+'e',s+'ons',s+'ez',ss+'ent'];}return[s+'e',s+'es',s+'e',s+'ons',s+'ez',s+'ent'];}
function ruleConj(inf){
  let s,pres,impf,fut,cond,subj,ps,imp;
  if(inf.endsWith('er')){s=inf.slice(0,-2);pres=erPres(inf);impf=[s+'ais',s+'ais',s+'ait',s+'ions',s+'iez',s+'aient'];const b=inf.endsWith('e')?inf.slice(0,-1):inf;fut=[b+'rai',b+'ras',b+'ra',b+'rons',b+'rez',b+'ront'];cond=buildCond(fut);const ss=pres[5].endsWith('ent')?pres[5].slice(0,-3):s;subj=[ss+'e',ss+'es',ss+'e',s+'ions',s+'iez',ss+'ent'];ps=[s+'ai',s+'as',s+'a',s+'âmes',s+'âtes',s+'èrent'];imp=[pres[1],pres[3],pres[4]];}
  else if(inf.endsWith('ir')){s=inf.slice(0,-2);const IR2=/(?:finir|choisir|réussir|remplir|agir|obéir|réfléchir|grandir|grossir|maigrir|rougir|blanchir|vieillir|punir|bâtir|nourrir|saisir|subir|accomplir|réunir|gémir|polir)$/.test(inf);if(IR2){pres=[s+'is',s+'is',s+'it',s+'issons',s+'issez',s+'issent'];impf=[s+'issais',s+'issais',s+'issait',s+'issions',s+'issiez',s+'issaient'];subj=[s+'isse',s+'isses',s+'isse',s+'issions',s+'issiez',s+'issent'];ps=[s+'is',s+'is',s+'it',s+'îmes',s+'îtes',s+'irent'];}else{const ss=s.slice(0,-1);pres=[ss+'s',ss+'s',ss+'t',s+'ons',s+'ez',s+'ent'];impf=[s+'ais',s+'ais',s+'ait',s+'ions',s+'iez',s+'aient'];subj=[s+'e',s+'es',s+'e',s+'ions',s+'iez',s+'ent'];ps=[s+'is',s+'is',s+'it',s+'îmes',s+'îtes',s+'irent'];}fut=[inf+'ai',inf+'as',inf+'a',inf+'ons',inf+'ez',inf+'ont'];cond=buildCond(fut);imp=[pres[1],pres[3],pres[4]];}
  else{s=inf.slice(0,-2);pres=[s+'s',s+'s',s+'',s+'ons',s+'ez',s+'ent'];impf=[s+'ais',s+'ais',s+'ait',s+'ions',s+'iez',s+'aient'];fut=[inf+'ai',inf+'as',inf+'a',inf+'ons',inf+'ez',inf+'ont'];cond=buildCond(fut);subj=[s+'e',s+'es',s+'e',s+'ions',s+'iez',s+'ent'];ps=[s+'is',s+'is',s+'it',s+'îmes',s+'îtes',s+'irent'];imp=[pres[1],pres[3],pres[4]];}
  return{présent:pres,imparfait:impf,futur_simple:fut,conditionnel_présent:cond,subjonctif_présent:subj,passé_simple:ps,impératif_présent:imp};
}

function conjugate(raw){
  const inf=raw.trim().toLowerCase();
  if(IRREG[inf])return{ok:true,conj:fillCompound(JSON.parse(JSON.stringify(IRREG[inf])),inf)};
  const pfx=['re','dé','é','in','sur','sous','pré','entre','con','par','pro','contra','contre','inter','extra','im','ir'];
  for(const p of pfx){if(inf.startsWith(p)&&IRREG[inf.slice(p.length)]){const base=inf.slice(p.length);const bc=JSON.parse(JSON.stringify(IRREG[base]));const pc={};for(const[t,fs]of Object.entries(bc))pc[t]=fs.map(f=>f==='—'?'—':p+f);return{ok:true,conj:fillCompound(pc,inf)};}}
  try{return{ok:true,conj:fillCompound(ruleConj(inf),inf)};}catch{return{ok:false,conj:{}};}
}

function buildQs(lid,mode){const l=lessons.find(l=>l.id===lid);const getTs=cid=>[...document.querySelectorAll(`#${cid} input:checked`)].map(i=>i.value);let qs=[];if(mode==='tense'){const ts=getTs('tense-checks');if(!ts.length){toast('请至少选一个时态');return null;}const verbs=l.verbs.filter(v=>verbData[v]?.ok);if(!verbs.length){toast('无已就绪动词');return null;}verbs.forEach(inf=>ts.forEach(tk=>{const forms=verbData[inf]?.conj?.[tk];if(!forms)return;const ps=PRON[tk]||PRON.présent;ps.forEach((pron,i)=>{const f=Array.isArray(forms)?forms[i]:'';if(f&&f!=='—')qs.push({inf,tk,pron,ans:f});});}));}else{const inf=document.getElementById('sel-verb').value;if(!inf||!verbData[inf]?.ok){toast('请选择有效动词');return null;}const ts=getTs('tense-checks-verb');if(!ts.length){toast('请至少选一个时态');return null;}ts.forEach(tk=>{const forms=verbData[inf]?.conj?.[tk];if(!forms)return;const ps=PRON[tk]||PRON.présent;ps.forEach((pron,i)=>{const f=Array.isArray(forms)?forms[i]:'';if(f&&f!=='—')qs.push({inf,tk,pron,ans:f});});});}if(!qs.length){toast('无法生成题目');return null;}return shuffle(qs);}
function showQ(){if(QI>=QQ.length){showResults();return;}const q=QQ[QI];const tense=TENSES.find(t=>t.key===q.tk);document.getElementById('q-tense').textContent=tense?.fr||q.tk;document.getElementById('q-pron').textContent=q.pron;document.getElementById('q-inf').textContent='('+q.inf+')';document.getElementById('answer-input').value='';const rb=document.getElementById('result-bar');rb.className='result-bar';rb.textContent='';document.getElementById('btn-next').style.display='none';document.getElementById('answer-input').focus();awaitNext=false;const tot=QQ.length;document.getElementById('q-counter').textContent=`${QI+1} / ${tot}`;document.getElementById('prog-fill').style.width=`${(QI/tot)*100}%`;document.getElementById('q-score').textContent=`${QC} 正确`;}

La bonne réponse : ${q.ans}`;document.getElementById('btn-next').style.display='inline-flex';document.getElementById('q-score').textContent=`${QC} 正确`;awaitNext=true;}

function norm(s){return s.trim().toLowerCase().replace(/[àâä]/g,'a').replace(/[éèêë]/g,'e').replace(/[îï]/g,'i').replace(/[ôö]/g,'o').replace(/[ùûü]/g,'u').replace(/ç/g,'c').replace(/\s+/g,' ');}
function shuffle(a){const b=[...a];for(let i=b.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[b[i],b[j]]=[b[j],b[i]];}return b;}
function esc(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');}
